65  bool CPlaceable::place(const Vector3 newOriginPosition){
66  // Don't do anything if the origin position is not changing (and tiles have been already initialitated)
67  if ((newOriginPosition == _floorOriginPosition) && _tiles.capacity()>0)
68  return true;
69 
70  // Check if tiles allow the placement
71  bool check = checkPlacementIsPossible(newOriginPosition);
72  if (!check)
73  return false;
74 
75  // Clean entity above in old tiles
76  _tileManager->getTile(_floorOriginPosition)->setEntityAbove(nullptr);
77  for (int x = 0; x < _floorX; ++x) {
78  for (int z = 0; z < _floorZ; ++z) {
79  Tile* tile = _tileManager->getTile(_floorOriginPosition + Vector3(x, 0, z));
80  tile->setEntityAbove(nullptr);
81  }
82  }
83 
84  // Store new origin position
85  _floorOriginPosition = newOriginPosition;
86 
87  // Clear tile vector
88  _tiles.clear();
89 
90  // Initialize vector to calculate the average position of all tiles
91  Vector3 centerPosition(0, 0, 0);
92 
93  // For each tile
94  for (int x = 0; x < _floorX; ++x) {
95  for (int z = 0; z < _floorZ; ++z) {
96  Tile* tile = _tileManager->getTile(_floorOriginPosition + Vector3(x, 0, z));
97 
98  // Store it internally
99  _tiles.push_back(tile);
100 
101  // Notify its new entity above
102  tile->setEntityAbove(this->getEntity());
103 
104  // Add it to the average position
105  centerPosition += tile->getEntity()->getPosition();
106  }
107  }
108 
109  // Get the average position
110  centerPosition /= _tiles.size();
111 
112  // Add some height to the average found so the object is over the tile
113  centerPosition += Vector3(0, HEIGHT_OVER_TILE, 0);
114 
115  // Move entity
116  _entity->setPosition(centerPosition);
117 
118  return true;
