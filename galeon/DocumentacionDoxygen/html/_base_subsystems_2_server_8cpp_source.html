179  bool CServer::initOgre()
180  {
181  // Inicializamos el punto de acceso a Ogre con una definición
182  // diferente de los plugins si estamos en release o en debug.
183 #ifdef _DEBUG
184  _root = new Ogre::Root("plugins_d.cfg");
185 #else
186  _root = new Ogre::Root("plugins.cfg");
187 #endif
188  // Cargamos las rutas de los recursos desde el archivo de configuración
189  setupOgreResources();
190 
191  // Cargamos los ajustes de configuración.
192  if(!configureOgre())
193  return false;
194 
195  // Inicializamos los recursos y el sistema de renderizado.
196  if(!initOgreResources())
197  {
198  // La unicialización puede haber fallado por cargar una
199  // configuración incorrecta, por haber actualizado hardware o
200  // cambiado de máquina. Mostramos la ventana de configuración.
201  if(!configureOgre(true)) {
202  // Si el juego no se puede configurar se finaliza
203  return false;
204  }
205  // Si tras actualizar la configuración sigue sin poderse desistimos.
206  if(!initOgreResources())
207  return false;
208  }
209  return true;
210 
211  } // initOgre

215  bool CServer::initOIS()
216  {
217  // Cogemos el identificador y las dimensiones de la ventana
218  // de renderizado para el gestor de periféricos de entrada.
219  // Para saber el espacio de actuación del ratón.
220  unsigned int width, height, windowHnd;
221  getWindowExtents(width,height);
222  if((width == -1) || (height == -1))
223  return false;
224 
225  if((windowHnd = getWindowHnd()) == -1)
226  return false;
227 
228  // Preparamos variables para la inicialización del sistema
229  // de entrada.
230  OIS::ParamList paramList;
231  std::ostringstream windowHndStr;
232 
233  windowHndStr << windowHnd;
234  paramList.insert(std::make_pair(std::string("WINDOW"), windowHndStr.str()));
235 
236 #if defined NON_EXCLUSIVE_MODE_IN_WINDOW_MODE
237  // Si estamos en modo ventana no queremos ratón y teclado en modo exclusivo,
238  // queremos que las demás aplicaciones puedan usarlo.
239  if(!_renderWindow->isFullScreen())
240  {
241 #if defined OIS_WIN32_PLATFORM
242  paramList.insert(std::make_pair(std::string("w32_mouse"), std::string("DISCL_FOREGROUND" )));
243  paramList.insert(std::make_pair(std::string("w32_mouse"), std::string("DISCL_NONEXCLUSIVE")));
244  paramList.insert(std::make_pair(std::string("w32_keyboard"), std::string("DISCL_FOREGROUND")));
245  paramList.insert(std::make_pair(std::string("w32_keyboard"), std::string("DISCL_NONEXCLUSIVE")));
246 #elif defined OIS_LINUX_PLATFORM
247  paramList.insert(std::make_pair(std::string("x11_mouse_grab"), std::string("false")));
248  paramList.insert(std::make_pair(std::string("x11_mouse_hide"), std::string("false")));
249  paramList.insert(std::make_pair(std::string("x11_keyboard_grab"), std::string("false")));
250  paramList.insert(std::make_pair(std::string("XAutoRepeatOn"), std::string("true")));
251 #endif
252  }
253 #endif
254 
255  // Creamos el sistema de entrada
256  try
257  {
258  _inputSystem = OIS::InputManager::createInputSystem(paramList);
259  }
260  catch(std::exception e)
261  {
262  return false;
263  }
264 
265  // Si es posible creamos el buffer del teclado.
266  if (_inputSystem->getNumberOfDevices(OIS::OISKeyboard) > 0)
267  {
268  _keyboard = static_cast<OIS::Keyboard*>(_inputSystem->createInputObject(OIS::OISKeyboard, true));
269  }
270 
271  // Si es posible creamos el buffer del ratón.
272  if (_inputSystem->getNumberOfDevices(OIS::OISMouse) > 0)
273  {
274  _mouse = static_cast<OIS::Mouse*>(_inputSystem->createInputObject(OIS::OISMouse, true));
275 
276  // Establecemos las dimensiones de la ventana de
277  // renderizado para saber el espacio de actuación
278  // del ratón.
279  setWindowExtents(width, height);
280  }
281 
282  return true;
283 
284  } // initOIS

405 
406  // Carga las rutas de los recursos desde el archivo de configuración
407  Ogre::ConfigFile cf;
408  cf.load("resources.cfg");
409 
410  // Examina todos los ajustes en el archivo
411  Ogre::ConfigFile::SectionIterator itSection = cf.getSectionIterator();
412 
413  std::string sSection, sType, sArch;
414  while(itSection.hasMoreElements())
415  {
416  sSection = itSection.peekNextKey();
417 
418  Ogre::ConfigFile::SettingsMultiMap *mapSettings = itSection.getNext();
419  Ogre::ConfigFile::SettingsMultiMap::const_iterator itSetting = mapSettings->begin();
420  while(itSetting != mapSettings->end())
421  {
422  sType = itSetting->first;
423  sArch = itSetting->second;
424 
425  Ogre::ResourceGroupManager::getSingleton().
426  addResourceLocation(sArch, sType, sSection);
427 
428  ++itSetting;
429  }
430  }
431 
432  } // setupResources
