69 void CCollisionManager::onTrigger(PxTriggerPair *pairs, PxU32 count)
70 {
71  // Recorrer el array de colisiones
72  for (unsigned int i=0; i<count; i++) {
73 
74  // Ignoramos los pares en los que alguna de las shapes (del trigger o de la otra entidad) ha sido borrada
75  if (pairs[i].flags & (PxTriggerPairFlag::eDELETED_SHAPE_TRIGGER | PxTriggerPairFlag::eDELETED_SHAPE_OTHER))
76  continue;
77 
78  // Comprobamos si estamos saliendo o entrando en el trigger
79  bool enter = pairs[i].status & PxPairFlag::eNOTIFY_TOUCH_FOUND;
80  bool exit = pairs[i].status & PxPairFlag::eNOTIFY_TOUCH_LOST;
81 
82  // Sólo tenemos en cuenta los eventos de entrada y salida del trigger. En PhysX 2.8 y anteriores
83  // también se notificada si el objeto permanecía dentro del trigger. La documentación no deja muy
84  // claro si este comportamiento se ha eliminado de manera definitiva.
85  if (!enter && !exit)
86  continue;
87 
88  // Obtener el componente lógico asociado al trigger físico
89  IPhysics *triggerComponent = (IPhysics *) pairs[i].triggerShape->getActor()->userData;
90  assert(triggerComponent);
91 
92  // Obtener el componente lógico asociado a la otra entidad física
93  IPhysics *otherComponent = (IPhysics *) pairs[i].otherShape->getActor()->userData;
94  assert(otherComponent);
95 
96  // Notificar a ambos componentes la colisión
97  triggerComponent->onTrigger(otherComponent, enter);
98  otherComponent->onTrigger(triggerComponent, enter);
99  }
100 }
