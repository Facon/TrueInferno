95  void CTileManager::loadInitialMatrix(CMap *map)
96  {
97  // Coge la Map::CEntity "Tile" leída del fichero de mapa a modo de prefab.
98  // @TODO Hacerlo en Map::CParser mediante una función genérica que reciba el nombre de la Map::CEntity.
99  Map::CMapParser::TEntityList mapEntityList =
100  Map::CMapParser::getSingletonPtr()->getEntityList();
101 
102  Map::CMapParser::TEntityList::const_iterator it, end;
103  it = mapEntityList.begin();
104  end = mapEntityList.end();
105 
106  // Itera sobre la lista de entidades del mapa hasta encontrar el prefab "Tile".
107  for (; it != end; it++) {
108  if ((*it)->getType() == "Tile") {
109  tileMapEntity = *it;
110  }
111  }
112 
113  assert(tileMapEntity && "Map::CEntity Tile not found");
114 
115  // Genera todas las Logic::CEntity tiles de la matriz a partir de la
116  // Map::CEntity "Tile" leída del fichero de mapa.
117  CEntityFactory* entityFactory = CEntityFactory::getSingletonPtr();
118 
119  std::string tileBaseName = tileMapEntity->getStringAttribute("name");
120  Vector3 tileBasePosition = tileMapEntity->getVector3Attribute("position");
121  Vector3 tileBaseDimensions = tileMapEntity->getVector3Attribute("dimensions");
122 
123  // Material base para los tiles.
124  std::string baseMaterialName = tileMapEntity->getStringAttribute("material");
125  // Material alternativo para los tiles.
126  // Se usa para diferenciar visualmente tiles adyacentes entre sí.
127  std::string altMaterialName = baseMaterialName + "_alt";
128 
129  // Reserva de memoria para la matriz de tiles.
130  _tiles = new Tile**[SIZE_X];
131  for (int x = 0; x < SIZE_X; ++x) {
132  _tiles[x] = new Tile*[SIZE_Z];
133  for (int z = 0; z < SIZE_Z; ++z) {
134  _tiles[x][z] = nullptr;
135  }
136  }
137 
138  // Lectura de los tiles del mapa.
139  // Se leen en el mismo orden en el que luego se les asigna el terreno, pero no es necesario.
140  for (int z = SIZE_Z - 1; z >= 0; --z) {
141  for (int x = SIZE_X - 1; x >= 0; --x) {
142  // Cambia la posición.
143  Vector3 tilePosition(tileBasePosition);
144 
145  tilePosition.x = (float)x; // *tileBaseDimensions.x;
146  tilePosition.y = (float)0; // Matriz de tiles 2D
147  tilePosition.z = (float)z; // *tileBaseDimensions.z;
148 
149  // Construye la nueva posición como un Vector3 representado por un
150  // string con formato: "x y z".
151  // @TODO Esto debería hacerse en Map::MapEntity.
152  std::stringstream newPosition;
153  newPosition << tilePosition.x << " " << tilePosition.y << " " << tilePosition.z;
154  tileMapEntity->setAttribute("position", newPosition.str());
155 
156  // Cambia el nombre (debe ser único!).
157  std::stringstream newTileName;
158  newTileName << tileBaseName;
159  newTileName << "_" << tilePosition.x << "_" << tilePosition.z;
160  tileMapEntity->setName(newTileName.str());
161 
162  // Crea una nueva Logic::CEntity tile.
163  CEntity *entityTile = entityFactory->createEntity(tileMapEntity, map);
164  assert(entityTile && "Failed to create entity Tile[X,Z]");
165  }

194  void CTileManager::loadTerrain(const std::string &filename) {
195  // Open file
196  std::ifstream file(filename);
197 
198  // Base material name for all tiles
199  std::string baseMaterialName = tileMapEntity->getStringAttribute("material");
200 
201  // For each line
202  std::string line;
203  int z = SIZE_Z-1;
204 
205  while (z>=0 && std::getline(file, line)) {
206  // Read line
207  std::stringstream ss(line);
208 
209  // For each token in line delimited by whitespace
210  int x = SIZE_X-1;
211  std::string item;
212 
213  while (x>=0 && std::getline(ss, item, ' ')) {
214  assert((item.size() == 1) && "Only tokens of 1 char are allowed in the terrain map");
215 
216  TerrainType terrainType;
217  char terrainTypeChar(item[0]);
218 
219  switch (terrainTypeChar) {
220  case '0':
221  terrainType = TerrainType::Empty;
222  break;
223  case '1':
224  terrainType = TerrainType::Mineral;
225  break;
226  case '2':
227  terrainType = TerrainType::Gas;
228  break;
229  default:
230  assert("Unknown terrain type char: " + terrainType);
231  }
232 
233  // Alternative material for odd tiles
234  std::string materialName(baseMaterialName);
235  materialName += terrainTypeChar;
236 
237  MaterialMessage message;
238  message._type = MessageType::SET_MATERIAL_NAME;
239  message._name = (((x + z) % 2 == 0) ? materialName : materialName + "_alt");
240 
241  message.Dispatch(*_tiles[x][z]->getEntity());
242  _tiles[x][z]->setTerrainType(terrainType);
243  --x;
244  } // while x
245  --z;
246  } // while z
247 
248  file.close();
249 
250  } // loadTerrain
